292 def _hash_xor(op_a, op_b):

344 return key

346 def _hash_e(self, k, data): 

347 internal = self._hash_xor(k, data) 

348 for i in range(12): 

349 internal = self._hash_s(internal) 

350 internal = self._hash_p(internal) 

351 internal = self._hash_l(internal) 

352 k = self._hash_get_key(k, i) 

353 internal = self._hash_xor(internal, k) 

354 return internal 

355 

356 def _hash_g(self, hash_h, hash_n, data): 

357 k = self._hash_xor(hash_n, hash_h) 

358 k = self._hash_s(k) 

359 k = self._hash_p(k) 

360 k = self._hash_l(k) 

361 

362 internal = self._hash_e(k, data) 

363 internal = self._hash_xor(internal, hash_h) 

364 result = self._hash_xor(internal, data) 

365 return result 

366 

367 def update(self, data): 

368 """Update the hash object with the bytes-like object. 

369 

370 Args: 

371 :data: The string from which to get the hash. Repeated calls are equivalent 

372 to a single call with the concatenation of all the arguments: m.update(a); 

373 m.update(b) is equivalent to m.update(a+b). 

374 """ 

375 data = self._buff + data 

376 self._num_block = len(data) // BLOCK_SIZE 

377 for i in range(0, self._num_block * BLOCK_SIZE, BLOCK_SIZE): 

378 block = data[i:i + BLOCK_SIZE] 

379 self._hash_h = self._hash_g(self._hash_h, self._hash_n, block) 

380 self._hash_n = self._hash_add_512(self._hash_n, CONST_V_512) 

381 self._hash_sigma = self._hash_add_512(self._hash_sigma, block) 

382 self._pad_block_size = BLOCK_SIZE - len(data) % BLOCK_SIZE 

383 if self._pad_block_size < BLOCK_SIZE: 

384 self._buff = data[-(BLOCK_SIZE - self._pad_block_size):] 

385 

386 def hash_final(self): 

387 """Completes the hash calculation after the data update. 

388 """ 

389 self._pad_block_size = BLOCK_SIZE - len(self._buff) 

390 internal = bytearray(BLOCK_SIZE) 

391 internal[1] = (((BLOCK_SIZE - self._pad_block_size) * 8) >> 8) & 0xff 

392 internal[0] = ((BLOCK_SIZE - self._pad_block_size) * 8) & 0xff 

393 if self._pad_block_size <= BLOCK_SIZE: 

394 self._buff = self._buff + b'\x01' 

395 self._buff = self._buff + b'\x00' * (self._pad_block_size - 1) 

396 

397 self._hash_h = self._hash_g(self._hash_h, self._hash_n, self._buff) 

398 self._hash_n = self._hash_add_512(self._hash_n, internal) 

399 self._hash_sigma = self._hash_add_512(self._hash_sigma, self._buff) 

400 self._hash_h = self._hash_g(self._hash_h, CONST_V_0, self._hash_n) 

401 self._hash_h = self._hash_g(self._hash_h, CONST_V_0, self._hash_sigma) 

402 

403 def get_hash(self): 

404 """Return the value of the _hasha_h attribute. 

405 """ 

406 return self._hash_h[-self.digest_size:] 

407 

408 def digest(self): 

409 """Return the digest of the data passed to the update() method so far. 

410 This is a bytes object of size digest_size. 

411 """ 

412 temp = self.copy() 

413 temp.hash_final() 

414 return temp.get_hash() 

415 

416 def hexdigest(self): 

417 """Return the digest of the data passed to the update() method so far. 

418 This is a double-sized string object (digest_size * 2). 

419 """ 

420 temp = self.copy() 

421 temp = self.copy() 

422 temp.hash_final() 

423 return ''.join(format(x, '02x') for x in temp.get_hash())

426 """Сlears the values of all class attributes. 

427 """ 

428 self._buff = b'' 

429 self._num_block = 0 

430 self._pad_block_size = 0

432 self._hash_n = bytearray(BLOCK_SIZE)
